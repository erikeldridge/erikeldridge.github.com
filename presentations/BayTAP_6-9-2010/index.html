Example: Weather data

  • Get the weather forecast for Sunnyvale
  • http://query.yahooapis.com/v1/public/yql?q=select%20*%20from%20weather.forecast%20where%20location%3D94089
  • select * from weather.forecast where location=94089

Code to make the request

var query = 'SELECT * FROM weather.forecast WHERE location=94089';
var url = "http://query.yahooapis.com/v1/public/yql?" 	
  + "q=" + encodeURIComponent( query )
  + "&callback=callback";

var Y = new YUI();

Y.use('node', function(Y){
  Y.Get.script(url);
});

function callback(response) {
  Y.log( response.query.results );
}
